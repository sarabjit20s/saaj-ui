## Usage

```tsx
import { Alert } from '@/components/ui/alert';

export function AlertDemo() {
  return (
    <Alert>
      <Alert.Icon name="clock" />
      <Alert.Content>
        <Alert.Title>No days available</Alert.Title>
        <Alert.Description>
          Days will appear here when more photos and videos are added to the
          library.
        </Alert.Description>
      </Alert.Content>
    </Alert>
  );
}
```
