## Usage

```tsx
import { Avatar } from '@/components/ui/avatar';

export function AvatarDemo() {
  return (
    <Avatar>
      <Avatar.Image src="https://avatar.iran.liara.run/public?username=John+Doe" />
      <Avatar.Fallback>
        <Avatar.Icon name="user" />
      </Avatar.Fallback>
    </Avatar>
  );
}
```
